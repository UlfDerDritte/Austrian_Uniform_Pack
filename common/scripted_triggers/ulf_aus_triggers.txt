austrian_clothes_trigger = {
	scope:character ?= {
		home_country ?= {
			OR = {
				AND = {
					c:AUS ?= this
					is_subject = no # preventing uniform conflicts if AUS becomes a subject of another country with puppet uniforms
				}
				top_overlord = {
					was_formed_from = AUS
				}
				is_subject_of = c:AUS
				was_formed_from = AUS
			}		
		}
	}
}

late_military_trigger = {
	OR = {
		has_technology_researched = bolt_action_rifles
		has_technology_researched = trench_works
	}
}

late_military_clothes_trigger = {
	scope:character ?= {
		owner ?= {
			late_military_trigger = yes
		}
	}
}

late_military_clothes_pop_trigger = {
	scope:pop ?= {
		owner ?= {
			OR = {
			    late_military_trigger = yes
			}
		}
	}
}
